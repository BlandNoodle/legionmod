# Tractors
country_event = {
	id = 999999901
	title = "EVTNAME999999901"
	desc = "EVTDESC999999901"
	
	trigger = {
		NOT = {
			has_country_modifier = country_tractors
		}
		invention = tractors
	}
	
	option = {
		name = "EVTOPTA999999901"
		add_country_modifier = {
			name = country_tractors
			duration = -1
		}
	}
}
